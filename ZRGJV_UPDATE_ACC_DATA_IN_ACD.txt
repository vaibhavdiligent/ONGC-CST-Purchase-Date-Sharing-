*---------------------------------------------------------------------*
*       FORM prepare_sel_screen_frame
*---------------------------------------------------------------------*
form prepare_sel_screen_frame.

  loop at screen.
    perform close_block using: p_ado 'ADC' space,
                               p_ado 'ADO' abap_true.
    perform close_block using: p_upo 'UPC' space,
                               p_upo 'UPO' abap_true.
  endloop.

  case gv_sscr_ucomm.
    when 'UCOMM_OPENADO'.                   "Open Block
      set cursor field 'S_BUDAT'.
    when 'UCOMM_OPENUPO'.                   "Open Block
      set cursor field 'P_VNCORR'.
    when others.
      set cursor field gv_cursor_field.
  endcase.

endform.

*---------------------------------------------------------------------*
*       FORM process_sel_screen_actions
*---------------------------------------------------------------------*
form process_sel_screen_actions.

  get cursor field gv_cursor_field.

  gv_sscr_ucomm = sy-ucomm. "sscrfields-ucomm.

* Auf die User-Commands reagieren
  case gv_sscr_ucomm.
    when 'UCOMM_OPENADO'.                   "Open Block
      clear p_ado.
    when 'UCOMM_CLOSEADO'.                  "Close Block
      p_ado = abap_true.

    when 'UCOMM_OPENUPO'.                   "Open Block
      clear p_upo.
    when 'UCOMM_CLOSEUPO'.                  "Close Block
      p_upo = abap_true.
  endcase.

endform.

*---------------------------------------------------------------------*
*       FORM close_block
*---------------------------------------------------------------------*
form close_block using value(iv_close_block) like p_ado
                       iv_modify_id          like screen-group1
                       iv_convert.

  if not iv_convert is initial.
    if iv_close_block is initial.
      iv_close_block = abap_true.
    else.
      clear iv_close_block.
    endif.
  endif.

  if ( screen-group1 = iv_modify_id )
  and ( not iv_close_block is initial ).
    screen-active = '0'.
    modify screen.
  endif.

endform.                               " CLOSE_BLOCK

  duration short.

  private section.

    methods:
      run_ok for testing,
      parallelization_handlers_ok for testing.

class ltc_test implementation.

  method run_ok.

    data: lv_subrc type sy-subrc,
          lo_mock  type ref to if_jva_mig_correct_acdoca.

    lo_mock ?= cl_abap_testdouble=>create( 'IF_JVA_MIG_CORRECT_ACDOCA' ).
    zCL_JVA_MIG_CORRECT_ACDOCA=>set_instance( lo_mock ).

    try.
        perform run_process.
      catch cx_root.
        lv_subrc = 4.
    endtry.

    cl_abap_unit_assert=>assert_equals( act = lv_subrc
                                        quit = if_aunit_constants=>quit-no
                                        exp = 0 ).

  endmethod.

  method parallelization_handlers_ok.

    data: lv_subrc              type sy-subrc,
          ls_before_rfc_imp     type spta_t_before_rfc_imp,
          ls_before_rfc_exp     type spta_t_before_rfc_exp,
          lt_rfcdata            type spta_t_indxtab,
          lt_failed_objects     type spta_t_failed_objects,
          lt_objects_in_process type spta_t_objects_in_process,
          lv_user_param         type c,
          lv_rfcsubrc           type sy-subrc,
          lv_rfcmsg             type spta_t_rfcmsg,
          ls_after_rfc_imp      type spta_t_after_rfc_imp,
          ls_after_rfc_exp      type spta_t_after_rfc_exp,
          ls_in_rfc_imp         type spta_t_in_rfc_imp,
          ls_in_rfc_exp         type spta_t_in_rfc_exp.

    go_correct_acdoca = zCL_JVA_MIG_CORRECT_ACDOCA=>get_instance( ).

    try.
        perform before_rfc using    ls_before_rfc_imp
                           changing ls_before_rfc_exp
                                    lt_rfcdata
                                    lt_failed_objects
                                    lt_objects_in_process
                                    lv_user_param.
      catch cx_root.
        lv_subrc = 4.
    endtry.

    cl_abap_unit_assert=>assert_equals( act = lv_subrc
                                        quit = if_aunit_constants=>quit-no
                                        exp = 0 ).


    clear lv_subrc.

    try.
        perform in_rfc using    ls_in_rfc_imp
                       changing ls_in_rfc_exp
                                lt_rfcdata.
      catch cx_root.
        lv_subrc = 4.
    endtry.

    cl_abap_unit_assert=>assert_equals( act = lv_subrc
                                        quit = if_aunit_constants=>quit-no
                                        exp = 0 ).

    clear lv_subrc.

    try.
        perform after_rfc using    lt_rfcdata
                                   lv_rfcsubrc
                                   lv_rfcmsg
                                   lt_objects_in_process
                                   ls_after_rfc_imp
                          changing ls_after_rfc_exp
                                   lv_user_param.
      catch cx_root.
        lv_subrc = 4.
    endtry.

    cl_abap_unit_assert=>assert_equals( act = lv_subrc
                                        quit = if_aunit_constants=>quit-no
                                        exp = 0 ).

  endmethod.

endclass.
